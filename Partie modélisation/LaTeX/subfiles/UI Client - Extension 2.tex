\subparagraph{Fenêtre International transfer}
\access{En cliquand sur le bouton \emph{International Transfer} de la fenêtre \emph{Details}.}
\content{les boutons \emph{Back} et \emph{Confirm}, les labels \emph{Amount}, \emph{Recipient}, \emph{IBAN}, \emph{Message}, \emph{Date}, \emph{SWIFT}, \emph{country}, \emph{City}, \emph{Street}, \emph{Number} 
et les champs de texte \emph{Amount}, \emph{Recipient}, \emph{IBAN}, \emph{Message}, \emph{Date},\emph{SWIFT}, \emph{country}, \emph{City}, \emph{Street}, \emph{Number}.}
\begin{itemize}
    \item \navbutton{Back}{Main screen}
    \item \navbutton{Confirm}{Enter PIN to confirm} si les informations sont correctes, on effectue le virement si la réponse de la fenêtre \emph{Enter PIN to confirm} est positive
    \item \textfield{Recipient}{le destinataire}
    \item \textfield{IBAN}{l'IBAN du destinataire}
    \item \textfield{Message}{la communication liée au virement}
    \item \textfield{Date}{la date de planification du virement}
    \item \textfield{SWIFT}{Code swift de la banque du bénéficiaire}
    \item \textfield{Country}{Pays du bénéficiaire}
    \item \textfield{City}{Ville du bénéficiaire}
    \item \textfield{Street}{Rue du bénéficiaire}
    \item \textfield{Number}{Numéro de maison du bénéficiaire}
\end{itemize}
