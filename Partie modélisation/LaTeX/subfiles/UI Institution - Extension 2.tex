\paragraph{Nouvelles fenêtres disponibles}
\begin{itemize}
    \item AddCountryFee
    \item CountryFee
    \item MldifyCountryFee
    \item AddSubAccount
\end{itemize}

\subparagraph{Fenêtre \emph{AddCountryFee}}
\access{En cliquant sur le bouton \emph{Add} de la fenêtre \emph{CountryFee}}
\content{Les boutons \emph{Back} et Confirm ainsi que les champs de texte \emph{CountryName} et Fee}
\begin{itemize}
    \item \navbutton{Back}{CountryFee}
    \item Le bouton \emph{Confirm} ajoute les frais sur ce pays dans la base de donnée et ramène à la fenêtre \emph{CountryFee}
    \item \textfield{CountryName}{Le nom du pays}
    \item \textfield{Fee}{Les frais ajouté à ce pays}.
\end{itemize}

\subparagraph{Fenêtre \emph{ModifyCountryFee}}
Cette fenêtre est globalement similaire à la fenêtre \emph{AddCountryFee}.

\subparagraph{Fenêtre \emph{CountryFee}}
\access{En cliquand sur le bouton \emph{Manage Country Fee} de la fenêtre \emph{Main screen}.}
\content{Les boutons \emph{Back},\emph{Add} et \emph{Modify} ainsi que la liste \emph{Countries}}
\begin{itemize}
    \item \navbutton{Back}{Main screen}
    \item \navbutton{Add}{AddCountryFee}
    \item \navbutton{Modify}{ModifyCountryFee}
    \item La liste \emph{Countries} affiche la liste des pays qui ont déjà des frais dans cette banque.
\end{itemize}

\subparagraph{AddSubAccount}
\access{En cliquand sur le bouton \emph{AddSubAccount} de la fenêtre \emph{Details}.}
\content{Les boutons \emph{Back} et \emph{Confirm}, les labels \emph{Client IBAN}, \emph{Client Name}, \emph{Select new Currency} et la comboBox \emph{Currencies}.}
\begin{itemize}
    \item \navbutton{Back}{Details}
    \item Le bouton \emph{Confirm} permet d'ajouter le sous compte et de revenir à la fenêtre \emph{Details}.
    \item La comboBox \emph{Currencies} permet de sélectionner la devise du nouveau subAccount.
\end{itemize}
